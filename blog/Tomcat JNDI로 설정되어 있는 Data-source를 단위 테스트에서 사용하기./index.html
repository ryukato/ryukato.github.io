Tomcat JNDIлЎң м„Өм •лҗҳм–ҙ мһҲлҠ” Data-sourceлҘј лӢЁмң„ н…ҢмҠӨнҠём—җм„ң мӮ¬мҡ©н•ҳкё°.

Tomcatкіј к°ҷмқҖ WASлҘј кё°л°ҳмңјлЎң мӢӨн–үлҗ  м• н”ҢлҰ¬мјҖмқҙм…ҳмқ„ к°ңл°ңн•  л•Ң, ліҙнҶө м• н”ҢлҰ¬мјҖмқҙм…ҳкіј м—°лҸҷн•  DatabaseлҘј WASм—җ DataSourceлЎң м•„лһҳмҷҖ к°ҷмқҙ context.xmlм—җ м •мқҳ л°Ҹ л“ұлЎқн•ҳм—¬ мӮ¬мҡ©н•ҳкІҢ лҗңлӢӨ.

мң„мҷҖ к°ҷмқҙ DataSourceлҘј м •мқҳн•ҳм—¬ мӮ¬мҡ©н•ҳкІҢ лҗҳл©ҙ, WASм—җм„ң м ңкіөн•ҳлҠ” Connection Poolмқ„ мӮ¬мҡ©н•  мҲҳ мһҲлҠ” мқҙм җмқҙ мһҲкі , JNDIлҘј мӮ¬мҡ©н•ҳм—¬ м• н”ҢлҰ¬мјҖмқҙм…ҳмқҳ DAO к°қмІҙ л“ұкіј к°ҷмқҙ DatabaseмҷҖ м—°лҸҷн•ҙм•ј н•ҳлҠ” к°қмІҙм—җм„ң DataSourceмҷҖ Connectionмқ„ мүҪкІҢ к°Җм ёмҷҖ мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

к·ёлҹ°лҚ° Spring-jdbc нҳ№мқҖ IBatisмҷҖ к°ҷмқҖ н”„л Ҳмһ„мӣҢнҒ¬лҘј мӮ¬мҡ©н•ҳм§Җ м•Ҡкі  м§Ғм ‘ JDBC мҪ”л”©мқ„ н•ҙм„ң classлҘј мһ‘м„ұн•ҳкі  н…ҢмҠӨнҠён•ҙм•ј н•ҳлҠ” кІҪмҡ°, WASм—җ м„ м–ёлҗҳм–ҙ мһҲлҠ” DataSourceлҘј м–ҙл–»кІҢ к°Җм ёмҷҖм•ј н• к№Ң? лӢЁмң„ н…ҢмҠӨнҠёлҘј н•ҳкё° мң„н•ҙ WASлҘј мӢӨн–үн•ҙм•ј н• к№Ң? лҢҖлӢөмқҖ к·ёл Үм§Җ м•ҠлӢӨ. к·ё мқҙмң лҠ” DataSourceк°Җ WASм—җ м„ м–ёмқҙ лҗҳм–ҙ мһҲм§Җл§Ң, м„ м–ёлҗҳм–ҙ мһҲлҠ” DataSourceлҠ” JNDIлҘј нҶөн•ҙм„ң к°Җм ёмҳ¬ мҲҳ мһҲкё°л•Ңл¬ёмқҙлӢӨ. JNDIлҠ” н‘ңмӨҖ мқён„°нҺҳмқҙмҠӨмқҙкі  Tomcatкіј к°ҷмқҖ WASлҠ” JNDIмқҳ н‘ңмӨҖм—җ л§һлҠ” кө¬нҳ„мІҙлҘј м ңкіөн•ңлӢӨ. н•ңлӢӨл§Ҳл””лЎң Tomcatкіј к°ҷмқҖ WASк°Җ н•ҙмЈјлҠ” кІғмІҳлҹј DataSourceлҘј JNDIлҘј нҶөн•ҙ м—°кІ° н•ҙмЈјкё°л§Ң н•ҳл©ҙ лҗңлӢӨ.

мҳҲлҘј л“Өл©ҙ, м•„лһҳмҷҖ к°ҷмқҙ JNDIлҘј нҶөн•ҙ DataSourceлҘј к°Җм ёмҳӨлҠ” JndiDataSourceManagerк°Җ мһҲлӢӨкі  н•ҳкі  JndiDataSourceManagerк°Җ DataSourceлҘј мһҳ к°Җм ёмҳӨлҠ”м§ҖлҘј н…ҢмҠӨнҠён•ҙм•јн•ңлӢӨкі  н•ҳмһҗ. к·ёлҰ¬кі  context.xmlмқҖ src/main/webapp/META-INF/м—җ мһҲлӢӨ.

import javax.naming.InitialContext;
import javax.naming.NamingException;
import javax.sql.DataSource;

public class JndiDataSourceManager {
    private String defaultDataSourceJndiName = "java:comp/env/todoDataSource";

    @Override
    public DataSource getDataSource(String jndiName) {
        try {
            return (DataSource) new InitialContext().lookup(jndiName);
        } catch (NamingException e) {
            throw new RuntimeException("Fail to get a data-source with jndi: " + jndiName);
        }
    }

    @Override
    public DataSource getDefaultDataSource() {
        return getDataSource(defaultDataSourceJndiName);
    }
}

мң„мҷҖ к°ҷмқҖ мғҒнҷ©м—җм„ң JndiDataSourceManagerлҘј н…ҢмҠӨнҠён•ҳкё° мң„н•ҙм„ , лӢӨмқҢмқҳ лқјмқҙлёҢлҹ¬лҰ¬к°Җ н•„мҡ”н•ҳлӢӨ. pom.xmlм—җ м•„лһҳмқҳ dependencyл“Өмқ„ м¶”к°Җн•ҳкі ,

JndiDataSourceManagerлҘј н…ҢмҠӨнҠён•  classмқё JndiDataSourceManagerTestлҘј мғқм„ұн•ңлӢӨ. к·ёлҰ¬кі  лӢӨмқҢкіј к°ҷмқҙ мҪ”л“ңлҘј мһ‘м„ұн•ҳл©ҙ лҗңлӢӨ.

import hthurow.tomcatjndi.TomcatJNDI;
import org.junit.After;
import org.junit.Before;
import org.junit.Test;

import javax.sql.DataSource;
import java.io.File;

import static org.hamcrest.CoreMatchers.is;
import static org.hamcrest.core.IsNull.notNullValue;
import static org.junit.Assert.assertThat;

public class JndiDataSourceManagerTest {
    private DataSourceManager jndiDataSourceManager;
    private TomcatJNDI tomcatJNDI;

    @Before
    public void setUp() {
        tomcatJNDI = new TomcatJNDI();
        tomcatJNDI.processContextXml(new File("src/main/webapp/META-INF/context.xml"));
        tomcatJNDI.start();
        jndiDataSourceManager = new JndiDataSourceManager();
    }

    @After
    public void tearDown() {
        tomcatJNDI.tearDown();
    }

    @Test
    public void test_getDataSourceUsingJndi() {
        DataSource dataSource = jndiDataSourceManager.getDataSource("java:comp/env/todoDataSource");
        assertThat("data source from jndi", dataSource, is(notNullValue()));
    }

    @Test
    public void test_defaultDataSource() {
        DataSource dataSource = jndiDataSourceManager.getDefaultDataSource();
        assertThat("data source from jndi", dataSource, is(notNullValue()));
    }

    @Test(expected = RuntimeException.class)
    public void test_getDataSource_with_not_defined_name() {
        jndiDataSourceManager.getDataSource("not-defined-jndi-name");
    }
}

лЁјм Җ setUpм—җм„ң TomcatJNDI мқёмҠӨн„ҙмҠӨлҘј мғқм„ұн•ҙм„ң WASм—җм„ң мӮ¬мҡ©н•  context.xmlмқ„ м§Җм •н•ҙ мӨҖлӢӨ. к·ёлҰ¬кі  start. лӢЁ tearDownм—җм„ң л°ҳл“ңмӢң TomcatJNDI мқёмҠӨн„ҙмҠӨмқҳ tearDownмқ„ нҳём¶ңн•ҙ мЈјм–ҙм•ј н•ңлӢӨ.

к·ёлҰ¬кі  лӮҳлЁём§Җ н…ҢмҠӨнҠё мҪ”л“ңл“Өмқ„ мһ‘м„ұн•ҳм—¬ мӢӨн–үн•ҳл©ҙ, DataSourceлҘј л°ӣм•„мҳӨлҠ” л©”м„ңл“ңл“Өм—җ лҢҖн•ң н…ҢмҠӨнҠёк°Җ м„ұкіөн•ҳлҠ” кұё ліј мҲҳ мһҲлӢӨ.

мқҙл ҮкІҢ WASмқҳ contextм—җ DataSourceк°Җ мһҳ м„ м–ёмқҙ лҗҳм–ҙ мһҲлҠ”м§Җ, м„ м–ёлҗҳм–ҙ мһҲлҠ” DataSourceлҘј мһҳ к°Җм ёмҳӨлҠ”м§ҖлҘј нҷ•мқён•ҳкё° мң„н•ҙ WASлҘј мӢӨн–үн•  н•„мҡ” м—Ҷмқҙ лӢЁмң„ н…ҢмҠӨнҠёл§Ң мӢӨн–үн•ҳм—¬ нҷ•мқён•  мҲҳ мһҲлӢӨ. WASлҘј лқ„мҡ°кі  нҺҳмқҙм§Җ нҳ№мқҖ apiлҘј нҳём¶ңн•ҳлҠ” л“ұмқҳ End2End н…ҢмҠӨнҠёлҘј н•  н•„мҡ” м—Ҷмқҙ к°„лӢЁнһҲ лӢЁмң„ н…ҢмҠӨнҠёлҘј нҶөн•ҙ нҷ•мқён•  мҲҳ мһҲлҠ” л°©лІ•.
